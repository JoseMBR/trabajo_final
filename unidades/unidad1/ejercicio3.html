Ejercicio 3 - Unidad 1

Determinar el número mayor de tres números

En este ejercicio, se deben ingresar tres números y se mostrará el mayor de ellos.

a) Ingresar los tres números y determinar el mayor


<script>
    function determinarMayor() {
        let num1 = prompt("Introduce el primer número:");
        let num2 = prompt("Introduce el segundo número:");
        let num3 = prompt("Introduce el tercer número:");
        
        num1 = parseInt(num1);
        num2 = parseInt(num2);
        num3 = parseInt(num3);

        let mayor = Math.max(num1, num2, num3);
        alert("El número mayor es: " + mayor);
    }
</script>
